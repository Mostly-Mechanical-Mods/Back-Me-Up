BMU_contender_score = {
	if = {
		limit = {
			exists = scope:pressed_claim
			this = { save_temporary_scope_as = score_supporter }
			scope:pressed_claim.var:claim_target = { save_temporary_scope_as = holder save_temporary_scope_as = holder_candidate }
			scope:pressed_claim.var:claimed_title = { save_temporary_scope_as = title }
		}
		add = BMU_contender_score_gui
	}
	else_if = {
		limit = {
			exists = scope:actor.var:pressed_claim_interaction
			this = { save_temporary_scope_as = score_supporter }
			scope:actor.var:pressed_claim_interaction.var:claim_target = { save_temporary_scope_as = holder save_temporary_scope_as = holder_candidate }
			scope:actor.var:pressed_claim_interaction.var:claimed_title = { save_temporary_scope_as = title }
		}
		add = BMU_contender_score_gui
	}
}

			scope:holder = {
				this = scope:score_supporter
				NOT = {
					has_relation_rival = scope:score_contender #Unless is rival.
				}
			}
		}
		add = {

			format = "FF_VALUE_FORMAT"

BMU_tooltip_feudal_holder_elector_vote_not_child = {
	if = {
		limit = {
			scope:holder = {
				this = scope:score_supporter
				NOT = {
					scope:score_contender = {
						is_child_of = scope:holder
					}
				}
			}
		}
		add = {
			value = {
				subtract = 50
				if = { #Even less so if not same House.
					limit = {
						exists = scope:score_supporter.house
						exists = scope:score_contender.house
						NOT = {
							scope:score_supporter.house = scope:score_contender.house
						}
					}
					subtract = 50
				}
				if = { #Even less so if not same Dynasty.
					limit = {
						exists = scope:score_supporter.dynasty
						exists = scope:score_contender.dynasty
						NOT = {
							scope:score_supporter.dynasty = scope:score_contender.dynasty
						}
					}
					subtract = 150
				}
			}
			desc = BMU_tooltip_feudal_holder_elector_vote_not_child
			format = "FF_VALUE_FORMAT"
		}
	}
}

BMU_tooltip_feudal_holder_elector_vote_infertile = {

		add = {
			value = -50
			desc = BMU_tooltip_feudal_holder_elector_vote_infertile
			format = "FF_VALUE_FORMAT"
		}
	}
}

BMU_tooltip_feudal_elector_vote_agnatic_woman = {
	if = {
		limit = {
			OR = {
				scope:title = { #Title might have different law than realm's.
					has_title_law = male_preference_law
				}
				AND = { #If title has neither the required law nor its opposites, check for realm law.
					scope:holder = { has_realm_law = male_preference_law }
					scope:title = {
						NOR = {
							has_title_law = female_preference_law
							has_title_law = equal_law
						}
					}
				}
				faith = { has_doctrine = doctrine_gender_male_dominated }
			}
			scope:score_contender = { is_female = yes }
		}
		add = {
			value = {
				subtract = 100
				if = { #Increase if both.
					limit = {
						OR = {
							scope:title = { #Title might have different law than realm's.
								has_title_law = male_preference_law
							}
							AND = { #If title has neither the required law nor its opposites, check for realm law.
								scope:holder = { has_realm_law = male_preference_law }
								scope:title = {
									NOR = {
										has_title_law = female_preference_law
										has_title_law = equal_law
									}
								}
							}
						}
						faith = { has_doctrine = doctrine_gender_male_dominated }
					}
					multiply = 1.5
				}
				if = { #Halve if talking about myself.
					limit = {
						this = scope:score_contender
						scope:score_contender = scope:holder_candidate
					}
					multiply = 0.5
				}
			}
			desc = BMU_tooltip_feudal_elector_vote_agnatic_woman
			format = "FF_VALUE_FORMAT"
		}
	}
}

BMU_tooltip_feudal_elector_vote_political_concerns = {
	if = {
		limit = {
			is_landed = yes
			exists = capital_county
			capital_county = { scope:title = { is_de_jure_liege_or_above_target = prev } }

			OR = { # If the current holder is reigning from within the title, there is no reason to believe a lower tier candidate would not.
				scope:holder = { 
					capital_county = {
						scope:title = {
							NOT = { is_de_jure_liege_or_above_target = prev }
						}
					}
				}
				scope:score_contender = {
					highest_held_title_tier >= scope:title.tier
				}
			}

			scope:score_contender = {
				is_ruler = yes
				is_vassal_of = scope:holder
				exists = capital_county
				NOT = { capital_county = { scope:title = { is_de_jure_liege_or_above_target = prev } } }
			}
		}
		add = {
			value = {
				subtract = 15
				if = { #Boost for Scandinavian Elective.
					limit = {
						scope:title = { has_title_law = scandinavian_elective_succession_law }
					}
					subtract = 15

BMU_tooltip_feudal_elector_vote_opinion_negative = { #Raw Opinion bonus (negative)
	if = {
		limit = {
			NOT = { this = scope:score_contender } #Not on yourself.
			save_temporary_opinion_value_as = {
				name = opinion_of_candidate
				target = scope:score_contender
			}
			scope:opinion_of_candidate <= low_negative_opinion
		}
		add = {
			value = {
				subtract = 15
				if = { #Even less so if lower opinion.
					limit = {
						scope:opinion_of_candidate <= medium_negative_opinion
					}
					subtract = 15 #-30 total
				}
				if = { #Even less so if lower opinion.
					limit = {
						scope:opinion_of_candidate <= high_negative_opinion
					}
					subtract = 20 #-50 total
				}
				if = { #Even less so if lower opinion.
					limit = {
						scope:opinion_of_candidate <= very_high_negative_opinion
					}
					subtract = 50 #-100 total
				}
				if = { #Even less so if rival.
					limit = {
						has_relation_rival = scope:score_contender
					}
					subtract = 50 #-150 total
				}
				if = { #Further increased for Tanistry.
					limit = {
						scope:title = { has_title_law = gaelic_elective_succession_law }
					}
					multiply = 1.5
				}
			}
			desc = BMU_tooltip_feudal_elector_vote_opinion_negative
			format = "FF_VALUE_FORMAT"
		}
	}
}

BMU_tooltip_feudal_elector_vote_opinion_positive = { #Raw Opinion bonus (positive)
	if = {
		limit = {
			NOT = {
				has_relation_rival = scope:score_contender #Completely removed for rivals.

				name = opinion_of_candidate
				target = scope:score_contender
			}
			scope:opinion_of_candidate >= low_positive_opinion
		}
		add = {
			value = {
				value = 15
				if = { #Even more so if higher opinion.
					limit = {
						scope:opinion_of_candidate >= medium_positive_opinion
					}
					add = 15 #30 total
				}
				if = { #Even more so if higher opinion.
					limit = {
						scope:opinion_of_candidate >= high_positive_opinion
					}
					add = 20 #50 total
				}
				if = { #Even more so if higher opinion.
					limit = {
						scope:opinion_of_candidate >= very_high_positive_opinion
					}
					add = 20 #70 total
				}
				if = { #Even more so if friend/lover.
					limit = {
						OR = {
							has_relation_friend = scope:score_contender
							has_relation_lover = scope:score_contender
						}

					add = 30 #100 total
				}
				if = { #Further increased for Tanistry.
					limit = {
						scope:title = { has_title_law = gaelic_elective_succession_law }
					}
					multiply = 1.5
				}
			}
			desc = BMU_tooltip_feudal_elector_vote_opinion_positive
			format = "FF_VALUE_FORMAT"
		}
	}
}
