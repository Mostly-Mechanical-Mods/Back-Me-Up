on_game_start_after_lobby = {
	on_actions = {
		BMU_compat
		BMU_on_game_start
	}
}

# Set global variable that tracks if mod is active, useful for compatibility
BMU_compat = {
	effect = {
		set_global_variable = {
			name = BMU_is_loaded
			value = 1.000
		}
		add_to_global_variable_list = {
			name = mods_loaded
			target = global_var:BMU_is_loaded
		}
		# set_global_variable = boolean
	}
}

BMU_on_game_start = {
	effect = {
		# debug_log = "FF_on_game_start"
		add_to_global_variable_list = { name = BMU_gui_sort_options target = flag:BMU_gui_sort_options_max_mil_power }
		add_to_global_variable_list = { name = BMU_gui_sort_options target = flag:BMU_gui_sort_options_score_dif }
		add_to_global_variable_list = { name = BMU_gui_sort_options target = flag:BMU_gui_sort_options_rank }
	}
}
