BMU_claim_story = {
	on_setup = {
		if = {
			limit = {
				debug_only = yes
				NOT = { exists = story_owner.var:claimed_title }
			}
			debug_log = "BMU_claim_story - no claimed title"
			debug_log_scopes = yes
		}
		add_to_global_variable_list = {
			name = BMU_claim_stories
			target = scope:story
		}
		set_variable = { name = claimed_title value = story_owner.var:claimed_title }
		set_variable = { name = claim_target value = story_owner }
		story_owner = {
			add_to_variable_list = {
				name = claimed_titles
				target = scope:story
			}
		}
		var:claimed_title = {
			every_claimant = {
				explicit = all
				scope:story = {
					add_to_variable_list = {
						name = claimants
						target = prev
					}
				}
				scope:story = {
					add_to_variable_list = {
						name = supporters
						target = prev
					}
				}
				if = {
					limit = { NOT = { this = scope:story.story_owner }}
					add_to_variable_list = {
						name = claim_support
						target = scope:story
					}
				}
			}
		}
		story_owner = {
			every_vassal = {
				scope:story = { add_to_variable_list = { name = supporters target = prev }}
			}
			every_realm_county = {
				every_de_jure_top_liege = {
					scope:story = { add_to_variable_list = { name = supporters target = prev }}
					if = {
						limit = { NOT = { this = scope:story.story_owner }}
						add_to_variable_list = {
							name = claim_support
							target = scope:story
						}
					}
				}
			}
		}
	}
	on_end = {
	}
	on_owner_death = {
	}
	effect_group = {
		months = 1
		triggered_effect = {
			trigger = {
				
			}
			effect = {
				
			}
		}
	}
}
